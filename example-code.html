    const delay = (ms) => 
      new Promise((resolve) => 
        setTimeout(resolve, ms));
  

    async function compress(input) {
      await delay(3000);
      console.log('2) compress');
      return(input.slice(0, 3));
    }

    async function encrypt(input) {
      await delay(2000);
      console.log('3) encrypt');
      return(input
        .split('').reverse().join(''));
    }

    async function runProgram(str){
      console.log('1a) runProgram');
      const a = await compress(str);
      console.log('1b) runProgram');
      const b = await encrypt(a);
      console.log('1c) runProgram');
      return b;
    }

    runProgram('the secret key is 123543')
      .then((result) => {
        console.log(result);
      })
